NON-TERMINALS

Start ::= DeployTransition
| NormalTransition
| <EOF>
DeployTransition ::= <UNDERSCORE> <LBRACKET> <RBRACKET> Participant ExternalCall DeployAction <LBRACKET> ( Assertion )? <RBRACKET> <IDWN> ( <PLUS> )?
NormalTransition ::= <IDWN> <LBRACKET> ( Assertion )? <RBRACKET> ParticipantExtra ExternalCall NormalAction <LBRACKET> ( Assertion )? <RBRACKET> <IDWN> ( <PLUS> )?
ExternalCall ::= <GT>
| <LT>
Participant ::= <IDWN> ( <COLON> <IDWN> )?
ParticipantExtra ::= Participant ( <MOREPART> Participant )*
DeployAction ::= <STARTLABEL> <LPAR> <IDWN> ( <COMMA> ParamList )? <RPAR>
NormalAction ::= <IDWN> OperationType <IDWN> <LPAR> ParamList <RPAR>
OperationType ::= ( <MINUS> | <DOT> ) 
//	ASSERTIONS - PARAMETERS
ParamList ::= ( Type <COLON> <IDWN> ( <COMMA> Type <COLON> <IDWN> )* )?
Type ::= ( <INT> | <BOOL> | <STRING> | <IDWN> | <SET> )
Assertion ::= LogicalExp
LogicalExp ::= ( BooleanExp | <OR> <LPAR> LogicalExp <COMMA> LogicalExp <RPAR> | <AND> <LPAR> LogicalExp <COMMA> LogicalExp <RPAR> | <NOT> LogicalExp )
BooleanExp ::= ( ArithmeticExp ( <EQ> ArithmeticExp | <NEQ> ArithmeticExp | <LT> ArithmeticExp | <GT> ArithmeticExp | <LEQ> ArithmeticExp | <GEQ> ArithmeticExp | <VARASS> ArithmeticExp | <CONTAINS> ArithmeticExp )* )
| <EXISTS> AtomicExp AtomicExp LogicalExp
| <FORALL> AtomicExp AtomicExp LogicalExp
ArithmeticExp ::= UnaryExp ( <PLUS> UnaryExp | <MINUS> UnaryExp | <TIMES> UnaryExp | <DIV> UnaryExp )* 
/*
	Atomic and negation
*/
UnaryExp ::= ( AtomicExp | <MINUS> UnaryExp )
AtomicExp ::= ( <NUMBER> | <QUOTE> <STRING_LITERAL> <QUOTE> | <SLPAR> <SRPAR> | <TRUE> | <FALSE> | ( Type )? <IDWN> )
